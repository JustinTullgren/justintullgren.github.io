fun reverseWords(
    message: CharArray
) {
    if (message.size <= 1) return
    
    var left = 0
    var right = message.lastIndex
    
    while(left < right) {
        val temp = message[left]
        message[left] = message[right]
        message[right] = temp
        left++
        right--
    }
    
    left = 0
    while(left <= message.lastIndex) {
        var nextSpace = left
        while(nextSpace <= message.lastIndex && message[nextSpace] != ' ') {
            nextSpace++
        }
        right = nextSpace - 1
        while (left < right) {
            val temp = message[left]
            message[left] = message[right]
            message[right] = temp
            left++
            right--
        }
        left = nextSpace + 1
    }
    
}

fun test(a: String, b: String) {
    val chars = a.toCharArray()
    reverseWords(chars)
    val result = chars.joinToString("")
    val expected = b
    println("$result | $expected")
    if (result != expected) {
        throw Exception()
    }
}

fun main(args: Array<String>) {
    test("abc", "abc")
    test("abc def", "def abc")
    test("abc def hig", "hig def abc")
    test("abcd efg", "efg abcd")
        
}
